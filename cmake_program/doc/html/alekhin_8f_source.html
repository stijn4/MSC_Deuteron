2 c subroutine sfincldis_abkm09(xb,q2,PDFS,DPDFS,IPAR,ICOL)
3 c
4 c this is a code accessing the grid for dis inclusive structure
5 c functions(sfs). at the momentum transfer squared q2>1 gev^2
6 c the sfs are calculated in pqcd using the nnlo pdfs taken in the
7 c 3-flavour scheme with account of the qcd corrections to the
8 c massless wilson coefficients up to the nnlo. the c- and b-quark
9 c contributions are calculated in the 3-flavour scheme with account of
10 c the full nlo corrections and partial nnlo corrections due to the
11 c threshold resummation(cf. ref.[plb 672, 166 (2009)]).
12 c the target-mass corrections by georgi-politzer and the
13 c dynamical high-twist(ht) contribution are taken into account.
14 c the pdfs and the ht terms for the electromagnetic sfs
15 c were fitted to the inclusive dis non-resonant data down to
16 c w=1.8 gev and q2=2.5 gev^2 (cf. ref.[prd 81, 014032 (2010)]).
17 c the sfs are given in the kinematics, which corresponds to the
18 c c.m.s. energy less than 2e5 gev and momentum transferred squared
19 c bigger than 0.6 gev^2, outside of this region they are set to 0.
20 c the full electromagnetic sfs, which include the leading-twist
21 c terms corrected for the target-mass effects and the ht terms
22 c come from a smooth interpolation between the pqcd calculations
23 c at q2>1 gev^2 and the constraints defined by the current
24 c conservation at q2=0 (cf. ref.[arxiv:0710.0124]). the leading twist
25 c sfs, with and without account of the target-mass corrections,
26 c calculated in the pqcd through the whole grid kinematics are
27 c also provided, for completeness.
28 
29 c the corrections on nuclear effects in deuterium used in the
30 c analysis of ref.[prd 81, 014032 (2010)] are also provided, for
31 c completeness. they are calculated with the model of
32 c ref.[npa 765,126 (2006)] using the pdfs stored in this grid.
33 
34 c input parameters:
35 c xb is the parton momentum fraction
36 c q2 is the momentum transferred. the renormalization/factorization scale
37 c is set to q2 for the massless case and sqrt(q2+4m_h^2) for the
38 c heavy-quark contribution, where m_h is the heavy-quark mass.
39 c ipar controls output of the pdfs and \alpha_s uncertainties(see
40 c description of the output parameters below)
41 c icol selects the collision type (the beam, final state, and current)
42 c icol=1 -- the charged-lepton neutral current(photon exchange only)
43 c icol=2 -- the neutrino charged current
44 c icol=3 -- the anti-neutrino charged current
45 c icol=4 -- the neutrino/anti-neutrino neutral current
46

49 c and the sfs at given x and q:

55 c pdfs(3) -- f_2(p), leading twist + tmc + twist-4
56 
57 c pdfs(4) -- f_2(n), leading twist

59 c pdfs(6) -- f_2(n), leading twist + tmc + twist-4
60

63 c pdfs(9) -- f_t(p), leading twist + tmc + twist-4

71 c pdfs(15) -- xf_3(p), leading twist + tmc + twist-4

73 c pdfs(16) -- xf_3(n), leading twist

77 c pdfs(19) -- the correction on nuclear effects in deuterium for f_2

79 c pdfs(21) -- the correction on nuclear effects in deuterium for xf_3

81 c where (p) means proton, (n) -- neutron, and f_t=2xf_1.
82 c at the moment the structure functions xf_3 are set to 0.
83 
84 c output array dpdfs(0:np,nvar) contains derivatives of \alpha_s and
85 c the sfs on the parameters corresponding to the independent
86 c sources of the uncertainties, nvar is the number of these sources,
87 c equal to 46 in the current version. the input parameter ipar is used to
88 c optimize performance of the code. If ipar=0, no
89 c uncertainties are returned in dpdfs; if 0<ipar<=nvar, only
90 c the uncertainty due to the ipar-th source is returned; if ipar<0
91 c all uncertainties for the sources from 1 to nvar are returned.
92 c using derivatives stored in dpdfs one can take into account the
93 c correlations between different sfs and between sfs and \alpha_s.
94 c all derivatives are transformed to the orthonormal basis of eigenvectors

112 c s=s+(2*rndm(xxx)-1)/sqrt(32.)

315  SUBROUTINE spline(N,X,Y,B,C,D)
316 * ---------------------------------------------------------------------
317 * calculate the coefficients b,c,d in a cubic spline interpolation.
318 * interpolation subroutines are taken from
319 * g.e. forsythe, m.a. malcolm and c.b. moler,
320 * computer methods for mathematical computations(prentice-hall, 1977).
321 *
322  IMPLICIT REAL*8(a-h,o-z)
323 
324  dimension x(n), y(n), b(n), c(n), d(n)
325 *
326  nm1=n-1
327  IF(n.LT.2) RETURN
328  IF(n.LT.3) GO TO 250
329  d(1)=x(2)-x(1)
330  c(2)=(y(2)-y(1))/d(1)
331  DO 210 k=2,nm1
332  d(k)=x(k+1)-x(k)
333  b(k)=2.0d0*(d(k-1)+d(k))
334  c(k+1)=(y(k+1)-y(k))/d(k)
335  c(k)=c(k+1)-c(k)
336  210 CONTINUE
337  b(1)=-d(1)
338  b(n)=-d(n-1)
339  c(1)=0.0d0
340  c(n)=0.0d0
341  IF(n.EQ.3) GO TO 215
342  c(1)=c(3)/(x(4)-x(2))-c(2)/(x(3)-x(1))
343  c(n)=c(n-1)/(x(n)-x(n-2))-c(n-2)/(x(n-1)-x(n-3))
344  c(1)=c(1)*d(1)**2.0d0/(x(4)-x(1))
345  c(n)=-c(n)*d(n-1)**2.0d0/(x(n)-x(n-3))
346  215 CONTINUE
347  DO 220 k=2,n
348  t=d(k-1)/b(k-1)
349  b(k)=b(k)-t*d(k-1)
350  c(k)=c(k)-t*c(k-1)
351  220 CONTINUE
352  c(n)=c(n)/b(n)
353  DO 230 ib=1,nm1
354  k=n-ib
355  c(k)=(c(k)-d(k)*c(k+1))/b(k)
356  230 CONTINUE
357  b(n)=(y(n)-y(nm1))/d(nm1)
358  1 +d(nm1)*(c(nm1)+2.0d0*c(n))
359  DO 240 k=1,nm1
360  b(k)=(y(k+1)-y(k))/d(k)
361  1 -d(k)*(c(k+1)+2.0d0*c(k))
362  d(k)=(c(k+1)-c(k))/d(k)
363  c(k)=3.0d0*c(k)
364  240 CONTINUE
365  c(n)=3.0d0*c(n)
366  d(n)=d(n-1)
367  RETURN
368  250 CONTINUE
369  b(1)=(y(2)-y(1))/(x(2)-x(1))
370  c(1)=0.0d0
371  d(1)=0.0d0
372  b(2)=b(1)
373  c(2)=0.0d0
374  d(2)=0.0d0
375  RETURN
376  END
