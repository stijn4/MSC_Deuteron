22  FUNCTION f2p_b(xp,q2)
23  pm = 0.938272
24  f2p_b = 0.
25  yn = q2/(2.0*pm*xp)
26  fm2 = -q2 + 2.0*pm*yn + pm**2
27  IF(xp.GT.1..OR.fm2.LT.pm**2)return
28  fm = sqrt(fm2)
29  wwi = (2.*yn*pm+1.642)/(q2+0.376)
30  t = 1.-1./wwi
31  gw = 0.256*t**3+2.178*t**4+0.898*t**5-6.716*t**6+3.756*t**7
32  f2p_b = b(fm,q2)*gw*wwi*xp
33  RETURN
34  END

66  IF(wm.LE.0.94)RETURN

68  omega=1.+(wsq-pmsq)/qsq
69  x=1./omega
70  xpx=c(22)+c(23)*(x-c(24))**2
71  piemsq=(c(1)-pm)**2
72 ********************************************************
73 * added part
74 ********************************************************
75  b1 = 0.0
76  IF(wm.EQ.c(1))GOTO 11

98  DO 30 i=1,nres
99  index=(i-1)*3+1+nbkg

101  IF(i.EQ.1)ram=c(index)+c(18)*qsq+c(19)*qsq**2

131  function f2_wtl(ia,x,Q2)
132 *********************************************************************
133 * f2 Structure function of proton (ia=1) and deuteron (ia=2)
134 * within the Whitlow parameterization. From Phys.Lett. B282 (1992)475
135 * The Kinematical ranges are:x >= 0.062, Q^2>=0.6, nu<19 GeV,
136 * w^2> 3 GeV^2
137 **********************************************************************
138  dimension cw_p(12),cw_d(12)
139  data cw_p/1.417, -0.108, 1.486, -5.979, 3.524, -0.011,
140  & -0.619, 1.385, 0.270, -2.179, 4.722, -4.363/
141  data cw_d/0.948, -0.115, 1.861, -4.733, 2.348, -0.065,
142  & -0.224, 1.085, 0.213, -1.687, 3.409, -3.255/
143 
144  pm = 0.938379
145  aha = 1.22*exp(3.2*x)
146  ab = 20.0
147  ab1 = 7.7*(1.0/x +pm**2/q2-1.0)
148  if(ab1.lt.ab)ab=ab1
149  if(ia.eq.1)then
150  f2_thr = 0.0
151  do j = 1,5
152  f2_thr = f2_thr + cw_p(j)*(1.0-x)**(j+2)
153  enddo
154  alm1 = cw_p(9) + cw_p(10)*x + cw_p(11)*x**2 + cw_p(12)*x**3
155  alm2 = 0.0
156  if(q2.le.aha)alm2 = cw_p(6) + cw_p(7) *x + cw_p(8)*x**2
157  beta = 1.0
158  elseif(ia.eq.2)then
159  f2_thr = 0.0
160  do j = 1,5
161  f2_thr = f2_thr + cw_d(j)*(1.0-x)**(j+2)
162  enddo
163  alm1 = cw_d(9) + cw_d(10)*x + cw_d(11)*x**2 + cw_d(12)*x**3
164  alm2 = 0.0
165  if(q2.le.aha)alm2 = cw_d(6) + cw_d(7) *x + cw_d(8)*x**2
166  beta = 1.0/(1.0-exp(-ab))
167  endif
168  fc = 1.0 + alm1*alog(q2/aha) + alm2*alog(q2/aha)**2
169  f2_wtl = beta * f2_thr * fc
170  return
171  end

179  double precision function f2_ctq(in,x,q2)
180 ***********************************************
181 * in = 1 - proton
182 * in = 0 - neutron
183 *
184 * The range of (x, Q) used in this round of global analysis is, approxi-
185 * mately, 0.01 < x < 0.75 ; and 4 GeV^2 < Q^2 < 400 GeV^2 for fixed target
186 * experiments and 0.0001 < x < 0.1 from HERA data.
187 *
188 ***********************************************
189  double precision, intent(in) :: x,q2
190  integer, intent(in) :: in
191  dimension pdf(-6:6)
192  double precision xx,qq,pdf
193  f2_ctq = 0.0
194  if(x.ge.1.0)return
195  xx = x
196  q = sqrt(q2)
197  qq = q
198  iset = 1
199  call ctq3pds(iset,pdf,xx,qq,irt)
200  u = pdf(1) !+ Pdf(-1)
201  ub = pdf(-1)
202  d = pdf(2) !+ Pdf(-2)
203  db = pdf(-2)
204  s = pdf(3)
205  sb = pdf(-3)
206  if(in.eq.1)f2 = (4.0*(u+ub)+(d+db)+(s+sb))/9.0
207  if(in.eq.0)f2 = (4.0*(d+db)+(u+ub)+(s+sb))/9.0
208  f2_ctq = f2
209  return
210  end

216 C Version 3 CTEQ distribution function in a parametrized form.
217 
218 C By: H.L. Lai, J. Botts, J. Huston, J.G. Morfin, J.F. Owens, J. Qiu,
219 C W.K. Tung & H. Weerts; Preprint MSU-HEP/41024, CTEQ 404
220 
221 C This file contains three versions of the same CTEQ3 parton distributions:
222 C
223 C Two "front-end" subprograms:
224 C FUNCTION Ctq3Pd (Iset, Iparton, X, Q, Irt)
225 C returns the PROBABILITY density for a GIVEN flavor;
226 C SUBROUTINE Ctq3Pds(Iset, Pdf, XX, QQ, Irt)
227 C returns an array of MOMENTUM densities for ALL flavors;
228 C One lower-level subprogram:
229 C FUNCTION Ctq3df (Iset, Iprtn, XX, QQ, Irt)
230 C returns the MOMENTUM density of a GIVEN valence or sea distribution.
231 
232 C One supplementary function to return the QCD lambda parameter
233 C concerning these distributions is also included (see below).
234 
235 C Although DOUBLE PRECISION is used, conversion to SINGLE PRECISION
236 C is straightforward by removing the
237 C Implicit Double Precision statements.
238 
239 C Since this is an initial distribution of version 3, it is
240 C useful for the authors to maintain a record of the distribution
241 C list in case there are revisions or corrections.
242 C In the interest of maintaining the integrity of this package,
243 C please do not freely distribute this program package; instead, refer
244 C any interested colleagues to direct their request for a copy to:
245 C Lai@cteq11.pa.msu.edu or Tung@msupa.pa.msu.edu.
246 
247 C If you have detailed questions concerning these CTEQ3 distributions,
248 C or if you find problems/bugs using this initial distribution, direct
249 C inquires to Hung-Liang Lai or Wu-Ki Tung.
250 
251 C -------------------------------------------
252 C Detailed instructions follow.
253 
254 C Name convention for CTEQ distributions: CTEQnSx where
255 C n : version number (currently n = 3)
256 C S : factorization scheme label: = [M L D] for [MS-bar LO DIS]
257 c resp.
258 C x : special characteristics, if any
259 C (e.g. S(F) for singular (flat) small-x, L for "LEP lambda value")
260 C (not applicable to CTEQ3 since only three standard sets are given.)
261 
262 C Explanation of functional arguments:
263 
264 C Iset is the set label; in this version, Iset = 1, 2, 3
265 C correspond to the following CTEQ global fits:
266 
267 C cteq3M : best fit in the MS-bar scheme
268 C cteq3L : best fit in Leading order QCD
269 C cteq3D : best fit in the DIS scheme
270 
271 C Iprtn is the parton label (6, 5, 4, 3, 2, 1, 0, -1, ......, -6)
272 C for (t, b, c, s, d, u, g, u_bar, ..., t_bar)
273 C *** WARNING: We use the parton label 2 as D-quark, and 1 as U-quark which
274 C might be different with your labels.
275 
276 C X, Q are the usual x, Q;
277 C Irt is a return error code (see individual modules for explanation).
278 C
279 C ---------------------------------------------
280 
281 C Since the QCD Lambda value for the various sets are needed more often than
282 C the other parameters in most applications, a special function
283 C Wlamd3 (Iset, Iorder, Neff) is provided
284 C which returns the lambda value for Neff = 4,5,6 effective flavors as well as
285 C the order these values pertain to.
286 
287 C ----------------------------------------------
288 C The range of (x, Q) used in this round of global analysis is, approxi-
289 C mately, 0.01 < x < 0.75 ; and 4 GeV^2 < Q^2 < 400 GeV^2 for fixed target
290 C experiments and 0.0001 < x < 0.1 from HERA data.
291 
292 C The range of (x, Q) used in the reparametrization of the QCD evolved
293 C parton distributions is 10E-6 < x < 1 ; 1.6 GeV < Q < 10 TeV. The
294 C functional form of this parametrization is:
295 
296 C A0 * x^A1 * (1-x)^A2 * (1 + A3 * x^A4) * [log(1+1/x)]^A5
297 
298 C with the A'coefficients being smooth functions of Q. For heavy quarks,
299 C a threshold factor is applied to A0 which simulates the proper Q-dependence
300 C of the QCD evolution in that region according to the renormalization
301 C scheme defined in Collins-Tung, Nucl. Phys. B278, 934 (1986).
302 
303 C Since this function is positive definite and smooth, it provides sensible
304 C extrapolations of the parton distributions if they are called beyond
305 C the original range in an application. There is no artificial boundaries
306 C or sharp cutoff's.
307 C ------------------------------------------------
308 
309  FUNCTION ctq3pd (Iset, Iparton, X, Q, Irt)
310 
311 C This function returns the CTEQ parton distributions f^Iset_Iprtn/proton
312 C --- the PROBABILITY density
313 
314 C (Iset, Iparton, X, Q): explained above;
315 
316 C Irt : return error code: see module Ctq3df for explanation.
317 
318  IMPLICIT DOUBLE PRECISION (a-h, o-z)
319 
320  ifl = iparton
321  jfl = abs(ifl)
322 C Valence
323  IF (ifl.Eq.1 .or. ifl.Eq.2) THEN
324  vl = ctq3df(iset, ifl, x, q, irt)
325  ELSE
326  vl = 0.
327  ENDIF
328 C Sea
329  sea = ctq3df(iset, -jfl, x, q, irt)
330 C Full (probability) Distribution
331  ctq3pd = (vl + sea) / x
332 
333  Return
334 C *************************
335  END

339 
340 C This function returns the CTEQ parton distributions xf^Iset_Iprtn/proton
341 C --- the Momentum density in array form
342 c
343 C (Iset, X, Q): explained in header comment lines;
344 
345 C Irt : return error code -- cumulated over flavors:
346 C see module Ctq3df for explanation on individual flavors.
347 C Pdf (Iparton);
348 C Iparton = -6, -5, ...0, 1, 2 ... 6
349 C has the same meaning as explained in the header comment lines.

353

356  if(i.le.0) then

361  endif

364 
365  Return

367  End
