51 // TDeuteron::Wavefunction //
52 // //
53 // This class represents a generic deuteron wavefunction in strangecalc. //
54 // //
55 // Different wavefunctions have different implementations. Wavefunction //
56 // deals with this by owning a concrete instance of the //
57 // TWavefunctionImplementation interface. //
58 // //
59 // For safety reasons, users are not allowed to create a wavefunction //
60 // of their own 'in code'. Therefore all meaningful constructors are //
61 // hidden. The named constructors are the only way of creating a //
62 // Wavefunction. //
63 // //
64 // The deuteron with fourvector
65 // BEGIN_LATEX
66 // p_{D} = (M_{D}, #vec{p}_{D} )
67 // END_LATEX
68 // is composed of two nucleons, labeled 1 and 2.
69 // We define
70 // BEGIN_LATEX
71 // p = #frac{p_{1}-p_{2}}{2}
72 // p_{D} = p_{1}+p_{2}
73 // END_LATEX
74 // and
75 // BEGIN_LATEX
76 // p_{1} = #frac{p_{D}}{2} + p
77 // p_{2} = #frac{p_{D}}{2} - p
78 // END_LATEX
79 //
80 // Unless explicitely specified, we evaluate the deuteron state in its
81 // restframe.
82 // //

796 FourVector<GammaStructure> TDeuteron::Wavefunction::Vertex(const FourVector<double>& p2,
797  const FourVector<double>& pD) const
798 {
799  // The covariant Blankenbecler and Cook vertex with particle 2 (4momentum p2) on-mass shell:
800  // BEGIN_LATEX
801  // #Gamma = F_{1} #gamma - #frac{F_{2}}{M_{N}} p - #frac{M_{N}-#gamma.p_{1}}{M_{N}} #left( F_{3} #gamma - #frac{F_{4}}{M_{N}} p #right).
802  // END_LATEX
803  // with BEGIN_LATEX p=#frac{p_{D}}{2}-p_{2} END_LATEX.
804  //
805  // The scalar form factors of the vertex are linked to the radial wave functions. See Eq.(46) in PRD 20(1979)2361.
806 
807  static double mass = (M_P + M_N)/2.; // average nucleon mass
808 
809  // momentum 4vector of particle 1
810  FourVector<double> p1 = pD - p2;
811 
812  // relative momentum 4vector
813  FourVector<double> p4vect = pD*(1./2.) - p2;
814 
815  // To find the scalar form factors we need the relative momentum and energy
816  // in the rest frame of the deuteron.
817  double ep;
818  double p;
819  if( (Abs(pD[1])<STRANGEUFLOW) && (Abs(pD[2])<STRANGEUFLOW) && (Abs(pD[3])<STRANGEUFLOW) ) {
820  // We are in the rest frame
821  ep = p2[0];
822  p = Sqrt( p2[1]*p2[1] + p2[2]*p2[2] + p2[3]*p2[3] );
823  } else {
824  TLorentzRotation boost(-pD[1]/pD[0],-pD[2]/pD[0],-pD[3]/pD[0]);
825  FourVector<double> p2CM = boost*p2;
826  ep = p2CM[0];
827  p = Sqrt( p2CM[1]*p2CM[1] + p2CM[2]*p2CM[2] + p2CM[3]*p2CM[3] );
828  }
829 
830  // vertex form factors: (46) in PRD 20(1979)2361.
831  double f1 = Pi()*Sqrt(2.*kMd)*(2*ep-kMd)
832  * ( GetUp(p) - GetWp(p)/Sqrt(2.) + mass/p*Sqrt(3./2.)*GetVTp(p) );
833 
834  double f2 = Pi()*Sqrt(2.*kMd)*(2*ep-kMd)
835  * ( mass/(ep+mass)*GetUp(p) + mass*(2*ep+mass)/Sqrt(2.)/p/p*GetWp(p)
836  + mass/p*Sqrt(3./2.)*GetVTp(p) );
837 
838  double f3 = Pi()*Sqrt(2.*kMd)*ep*mass/p*Sqrt(3./2.)*GetVTp(p);
839 
840  double f4 = -Pi()*Sqrt(2.*kMd)*mass*mass/kMd
841  * ( (2*ep-kMd)*(GetUp(p)/(ep+mass) - (ep+2.*mass)/Sqrt(2.)/p/p*GetWp(p))
842  + kMd*Sqrt(3.)/p*GetVSp(p) );
843 
844  static GammaStructure unity(1.); // unit 4x4 matrix
845 
846  // vertex:
847  return ( f1*GMU
848  - (f2/mass*p4vect)*unity
849  - (unity - (p1*(1./mass))*GMU) * (f3*GMU - (f4/mass*p4vect)*unity) );
850 }

853 complex<double> TDeuteron::Wavefunction::SphericalHarmonic(unsigned int l, int m,
854  double theta, double phi)
855 {
856  // Compute the (normalized) spherical harmonic:
857  // BEGIN_LATEX
858  // Y_{lm}(#theta,#phi) = #sqrt{#frac{(2l+1)}{4#pi}#frac{(l-m)!}{(l+m)!}} P_{l}^{m}(cos#theta)e^{im#phi}
859  // END_LATEX
860  // with l>=0 and |m|<=l
861  //
862  // The Condon-Shortley phase convention is used.
863 
864  // first calculate Y_{l|m}}(theta,phi=0)
865  double Ylm = ROOT::Math::sph_legendre(l,TMath::Abs(m),theta);
866 
867  // (-1)^m when m is negative
868  if( m<0 ) Ylm *= (1+2*(m%2));
869 
870  // Add the phi-dependent phase before returning
871  return Ylm * exp(complex<double>(0,m*phi));
872 }

914  int nucleon1Pol,
915  const TVector3& p) const
916 {
917  // Calculate the state of a deuteron in momentum space.
918  // deuteronPol is the deuteron polarization (times two)
919  // nucleon1pol fixes the polarization of the first nucleon (the one that interacts with the photon). (times two)
920  //nucleon2pol fixes spectator nucleon polarization (times two)
921 
922  // The Dnp vertex matrix element is
923  // BEGIN_LATEX
924  // #frac{1}{2E_{1}} #bar{u}(p_{2},m_{2})u(#dot{p}_{1},m_{1})#Gamma_{#mu}#phi^{#mu}_{D}(m_{D})
925  // = #sum_{L} u_{L}(|#vec{p}_{1}|) #sum_{m_{L},m_{S}} <Lm_{L}Sm_{S}|1m_{D}><#frac{1}{2}m_{1}#frac{1}{2} m_{2}|1m_{S}> Y_{Lm_{L}}(#hat{p}_{1})
926  // END_LATEX
927  // with BEGIN_LATEX #dot{p}_{1} = ( #sqrt{#vec{p}_{1}^{2}+M_{1}^{2}},#vec{p}_{1}) END_LATEX the on-shell 4-momentum of particle 1.
928  // The labelling particle 1 and 2 is arbitrary in the non-relativistic case (You can do the math).
929 // if(IsRelativistic()) {
930 // cerr << "ERROR in TDeuteron::Wavefunction::DeuteronPState(TSpinor::Polarization&,"
931 // << " TVector3&&): "
932 // << "has no meaning for relativistic wave functions.\n";
933 // exit(1);
934 // }
935 
936 
937  complex<double> state=0.;
938 // L=2, mL=mD-mS, S=1, mS=m1+m2
939  for(int mS=-1; mS<=1; ++mS)
940  state -=
941  ClebschGordan(4,deuteronPol-mS*2,2,mS*2,2,deuteronPol)
942  *ClebschGordan(1,nucleon1Pol,1,nucleon2Pol,2,mS*2)

947  ClebschGordan(0,0,2,deuteronPol,2,deuteronPol)
948  *ClebschGordan(1,nucleon1Pol,1,nucleon2Pol,2,deuteronPol)

1073  // deuteronPol is the deuteron polarization (times two)

1087  // L=0, mL=0, S=1, mS=mD
1088  state +=
1089  ClebschGordan(0,0,2,deuteronPol,2,deuteronPol)
1090  *ClebschGordan(1,nucleon1Pol,1,nucleon2Pol,2,deuteronPol)
1091  *SphericalHarmonic(0,0,r.Theta(),r.Phi())
1092  *Radial_r(0,r.Mag());
1093 
1094  // L=2, mL=mD-mS, S=1, mS=m1+m2
1095 // for(int mS=-1; mS<=1; ++mS)
1096  state -=
1097  ClebschGordan(4,deuteronPol-nucleon1Pol-nucleon2Pol,2,nucleon1Pol+nucleon2Pol,2,deuteronPol)
1098  *ClebschGordan(1,nucleon1Pol,1,nucleon2Pol,2,nucleon1Pol+nucleon2Pol)
1099  *SphericalHarmonic(2,(deuteronPol-nucleon1Pol-nucleon2Pol)/2,r.Theta(),r.Phi())
