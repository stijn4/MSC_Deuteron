12 // TYukawaPWF //
13 // //
14 // TYukawaPWF implements the abstract interface //
15 // TWavefunctionImplementation. //
16 // //
17 // This class parametrizes non-relativistic wavefunctions as a //
18 // discrete superposition of Yukawa-type terms. //
19 // This parametrization was first introduced in PLB 101(1981)139. //
20 // //
21 // TWavefunctionImplementation fixes the following conventions //
22 // concerning the units: //
23 // * position in [fm] //
24 // * momentum in [MeV] //
25 // * r-space wavefunctions in [fm^-1/2] //
26 // * p-space wavefunctions in [MeV^-3/2] //
27 // //
28 // BEGIN_LATEX
29 // u(r) = #sum^{N}_{i=1} C_{i} e^{-m_{i}r}
30 // w(r) = #sum^{N}_{i=1} D_{i} e^{-m_{i}r} #left(1+#frac{3}{m_{i}r}+#frac{3}{m^{2}_{i}r^{2}}#right)
31 // v_{t}(r) = v_{s}(r) = 0
32 // END_LATEX //
33 // with //
34 // BEGIN_LATEX
35 // m_{i} = #alpha + (i-1) m_{0}
36 // END_LATEX
37 // In momentum space (after Fourier transformation) the parametrization is
38 // BEGIN_LATEX
39 // u(p) = #left(#frac{2}{#pi}#right)^{1/2} #sum^{N}_{i=1} #frac{C_{i}}{p^{2}+m_{i}^{2}}
40 // w(p) = #left(#frac{2}{#pi}#right)^{1/2} #sum^{N}_{i=1} #frac{-D_{i}}{p^{2}+m_{i}^{2}}
41 // v_{t}(p) = v_{s}(p) = 0
42 // END_LATEX
43 // Imposing boundary conditions on the wavefunction in configuration //
44 // space, leads to the following constraints for the parameters: //
45 // BEGIN_LATEX
46 // #sum^{N}_{i} C_{i} = #sum^{N}_{i} D_{i} = #sum^{N}_{i} D_{i} m_{i}^{2} = #sum^{N}_{i} #frac{D_{i}}{m_{i}^{2}} = 0
47 // END_LATEX
48 // //

65 TYukawaPWF::TYukawaPWF(int order, double alpha, double m0,
66  double *c, double *d)
67 : TWavefunctionImplementation(0), fOrder(order), fAlpha(alpha), fM0(m0),
68  fC(new double[order]), fD(new double[order]), fM(NULL)
69 {
70  // Constructor
71  //
72  // order = number of terms N in the parametrization
73  // alpha = Parameter BEGIN_LATEX #alpha END_LATEX in BEGIN_LATEX m_{i} END_LATEX formula [fm^-1]
74  // m0 = Parameter BEGIN_LATEX m_{0} END_LATEX in BEGIN_LATEX m_{i} END_LATEX formula [fm^-1]
75  // c,d = arrays containing parameters. [fm^1/2]
76 
77  if(!c || !d) {
78  cerr << "Error in TYukawaPWF::TYukawaPWF(int,double*,double*): "
79  << "double arrays are NULL!\n";
80  exit(1);
81  }
82 
83  if(fOrder<1) {
84  cerr << "Error in TYukawaPWF::TYukawaPWF(int,double*,double*): "
85  << "order should be >=1!\n";
86  exit(1);
87  }
88 
89  for(int i=0; i<fOrder; ++i) {
90  fC[i] = c[i];
91  fD[i] = d[i];
92  }
93 
94  // C_n, D_n, D_(n-1) and D_(n-2) are fixed,
95  // to satify the boundary conditions
96  // Finding C_n is trivial
97  fC[fOrder-1] = 0.;
98  for(int i=0; i<fOrder-1; ++i) fC[fOrder-1] -= fC[i];
99 
100  // To find D_n, D_(n-1) and D_(n-2) we solve a 3x3
101  // set of equations.
102  // We use ROOT's LU decomposition solver method.
103  double sumD=0., sumDm=0., sumD_m=0.;
104  for(int i=0; i<fOrder-3; ++i) {
105  sumD += fD[i];
106  sumDm += fD[i]*Mass(i+1)*Mass(i+1);
107  sumD_m += fD[i]/Mass(i+1)/Mass(i+1);
108  }
109 
110  double elements[9] = {1/Mass(fOrder)/Mass(fOrder),
111  1/Mass(fOrder-1)/Mass(fOrder-1),
112  1/Mass(fOrder-2)/Mass(fOrder-2),
113  1.,1.,1.,
114  Mass(fOrder)*Mass(fOrder),
115  Mass(fOrder-1)*Mass(fOrder-1),
116  Mass(fOrder-2)*Mass(fOrder-2),
117  };
118  TMatrixD extMatrix(3,3,elements);
119  elements[0]=-1.*sumD_m;
120  elements[1]=-1.*sumD;
121  elements[2]=-1.*sumDm;
122  TVectorD sol(3,elements);
123  TDecompLU lu(extMatrix);
124  lu.SetTol(1e-16);
125  lu.Solve(sol);
126 
127  fD[fOrder-3] = sol[2];
128  fD[fOrder-2] = sol[1];
129  fD[fOrder-1] = sol[0];
130 }

253 {
254  // Mass formula:
255  // BEGIN_LATEX
256  // m_{i} = #alpha + (i-1) m_{0}

264 double TYukawaPWF::GetUp(double p) const
265 {
266  // L=0 wave in momentum space.
267  // p = momentum in [MeV]
268  // Return value in units [MeV^-3/2]
269 
270  double radial=0.;
271 
272  for(int i=0; i<fOrder; ++i)
273  radial += fC[i] / ( p*p/fgHbarc/fgHbarc + Mass(i+1)*Mass(i+1) );
274 
275  return radial * TMath::Sqrt(2./TMath::Pi()) / pow(fgHbarc,3/2.);
276 }

350 double TYukawaPWF::GetUr(double r) const
351 {
352  // L=0 wave in configuration space
353  // r = relative position in [fm]
354  // Return value in units [fm^-1/2]
355 
356  double radial=0.;

358  for(int i=0; i<fOrder; ++i)
359  radial += fC[i]*TMath::Exp(-1.*Mass(i+1)*r);
360 
361  return radial;

365 double TYukawaPWF::GetWr(double r) const
366 {
367  // L=2 wave in configuration space
368  // r = relative position in [fm]
369  // Return value in units [fm^-1/2]
370 
371  double radial=0.;
372 
373  // convergence for r->0 is really bad. We have verified that
374  // for r<10^-5 the wavefunction can be considered zero.
375  if( r<1e-5 )
376  return 0;
377 
378  for(int i=0; i<fOrder; ++i)
379  radial += fD[i]*TMath::Exp(-1.*Mass(i+1)*r)
380  * (1.+ 3./Mass(i+1)/r + 3/TMath::Power(Mass(i+1)*r,2.));
381 
382  return radial;
383 }
